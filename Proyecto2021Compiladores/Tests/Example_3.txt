program
{
    integer inc(integer x)
    {
        return x + 1;
    }
    integer get__int() extern;
    void print_int_(integer i) extern ;
    void main()
    {
        integer y;
        y = get_int();
        if (y == 1) then
        {
            print_int(y);
        }
        else
        {
            return print_int_(inc(y));
        }
    }
}
